public static int[] quickSort(int[] arr, int left, int right) {
    if (right - lefts + 1 <= 1) {
        return arr;
    }

    int pivot = arr[right];
    int j = left;

    for (int i = left; i < right; i++) {
        if (arr[i] < pivot) {
            int tmp = arr[j];
            arr[j] = arr[i];
            arr[i] = tmp;
            j++;
        }
    }

    arr[right] = arr[j];
    arr[j] = pivot;

    quickSort(arr, left, j - 1);
    quickSort(arr, j + 1, right);

    return arr;
}
